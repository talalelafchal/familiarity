Downloading HTML string into an NSString or Swift String variable in iOS

<h1>Swift 3.0</h1> <pre><code>guard let url = URL(string: "http://www.example.com") else { return } do { let html = try String(contentsOf: url) } catch { //handle `error` here } </code></pre> <h1>Swift 2.3</h1> <pre><code>guard let url = NSURL(string: "http://www.example.com") else { return } do { let html = try String(contentsOfURL: url) } catch { //handle `error` here } </code></pre> <h2>Objective-C</h2> <pre><code>NSString *url = @"http://www.example.com"; NSURL *urlRequest = [NSURL URLWithString:url]; NSError *err = nil; NSString *html = [NSString stringWithContentsOfURL:urlRequest encoding:NSUTF8StringEncoding error:&amp;err]; if(err) { //Handle } </code></pre>
